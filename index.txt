<script>
    // CONFIG
    const API_URL = "http://127.0.0.1:8000/chat";
    const CLEAR_URL = "http://127.0.0.1:8000/clear";
    
    // STATE
    let chatHistory = JSON.parse(localStorage.getItem('genuine_webber_chats')) || [];
    let currentSessionMessages = [];
    const messagesArea = document.getElementById('messagesArea');
    const userInput = document.getElementById('userInput');
    const sendButton = document.getElementById('sendButton');
    const sidebar = document.getElementById('sidebar');

    // INIT
    window.onload = () => {
        renderHistoryList();
        addMessageToUI("Hello! I am Genuine Webber. How can I help you today?", 'bot');
    };

    // --- SIDEBAR TOGGLE FUNCTION ---
    function toggleSidebar() {
        sidebar.classList.toggle('closed');
    }

    // --- COPY FUNCTION ---
    function copyText(text, btnElement) {
        navigator.clipboard.writeText(text).then(() => {
            // Visual Feedback
            const originalText = btnElement.innerText;
            btnElement.innerText = '✅';
            btnElement.style.color = '#10b981'; // Green
            
            setTimeout(() => {
                btnElement.innerText = '📋';
                btnElement.style.color = '';
            }, 1500);
        }).catch(err => {
            console.error('Failed to copy: ', err);
        });
    }

    async function sendMessage() {
        const text = userInput.value.trim();
        if (!text) return;

        addMessageToUI(text, 'user');
        currentSessionMessages.push({ role: 'user', content: text });
        
        userInput.value = '';
        userInput.disabled = true;
        sendButton.disabled = true;
        document.getElementById('typingIndicator').style.display = 'flex';
        messagesArea.scrollTop = messagesArea.scrollHeight;

        try {
            const response = await fetch(API_URL, {
                method: 'POST',
                headers: { 'Content-Type': 'application/json' },
                body: JSON.stringify({ message: text })
            });

            if (!response.ok) throw new Error("Server Error");
            const data = await response.json();

            addMessageToUI(data.response, 'bot', data.used_tool);
            currentSessionMessages.push({ role: 'bot', content: data.response, usedTool: data.used_tool });

        } catch (error) {
            addMessageToUI("Error: Could not connect to server.", 'bot');
        } finally {
            userInput.disabled = false;
            sendButton.disabled = false;
            document.getElementById('typingIndicator').style.display = 'none';
            userInput.focus();
            messagesArea.scrollTop = messagesArea.scrollHeight;
            
            if (currentSessionMessages.length === 2) saveCurrentSessionToHistory();
        }
    }

    function addMessageToUI(text, sender, usedTool = false) {
        const msgDiv = document.createElement('div');
        msgDiv.className = `message ${sender}`;
        
        const avatarLetter = sender === 'user' ? 'U' : 'G';
        const toolBadge = usedTool ? `<div class="tool-badge">🔍 Search Tool Used</div>` : '';

        // Create Copy Button (Only for text content)
        const copyBtn = document.createElement('button');
        copyBtn.className = 'copy-btn';
        copyBtn.innerHTML = '📋';
        copyBtn.title = 'Copy text';
        copyBtn.onclick = () => copyText(text, copyBtn);

        msgDiv.innerHTML = `
            <div class="avatar">${avatarLetter}</div>
            <div class="msg-content">
                ${toolBadge}
                ${text.replace(/\n/g, '<br>')}
            </div>
        `;
        
        // Append copy button to the message bubble
        msgDiv.appendChild(copyBtn);
        messagesArea.appendChild(msgDiv);
    }

    async function startNewChat() {
        if (currentSessionMessages.length > 0) saveCurrentSessionToHistory();
        try { await fetch(CLEAR_URL, { method: 'POST' }); } catch (e) {}
        
        messagesArea.innerHTML = '';
        currentSessionMessages = [];
        addMessageToUI("Started a new conversation.", 'bot');
        
        // On mobile/small screens, close sidebar after clicking new chat
        if (window.innerWidth < 768) sidebar.classList.add('closed');
    }

    function saveCurrentSessionToHistory() {
        if (currentSessionMessages.length < 2) return;
        const summary = currentSessionMessages[0].content.substring(0, 25) + "...";
        const newSession = { id: Date.now(), title: summary, messages: [...currentSessionMessages] };
        chatHistory.unshift(newSession);
        if (chatHistory.length > 20) chatHistory.pop();
        localStorage.setItem('genuine_webber_chats', JSON.stringify(chatHistory));
        renderHistoryList();
    }

    function loadHistory(id) {
        const session = chatHistory.find(c => c.id === id);
        if (!session) return;
        messagesArea.innerHTML = '<center style="color:#9ca3af; margin:10px; font-size:0.8rem">Viewing Past Chat (Read-Only)</center>';
        session.messages.forEach(m => addMessageToUI(m.content, m.role, m.usedTool));
        currentSessionMessages = [];
        // Close sidebar on mobile when selecting a chat
        if (window.innerWidth < 768) sidebar.classList.add('closed');
    }

    function renderHistoryList() {
        const list = document.getElementById('historyList');
        list.innerHTML = '';
        chatHistory.forEach(chat => {
            const item = document.createElement('div');
            item.className = 'history-item';
            item.innerHTML = `<span>💬</span> ${chat.title}`;
            item.onclick = () => loadHistory(chat.id);
            list.appendChild(item);
        });
    }

    userInput.addEventListener('keypress', (e) => {
        if (e.key === 'Enter') sendMessage();
    });
</script>
